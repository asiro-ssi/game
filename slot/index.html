function startSpin() { if (spinning || coins < 3) return; coins -= 3; updateCoinDisplay(); document.getElementById("result").textContent = "回転中…"; stopped.fill(false); spinning = true; reels.forEach((reel, index) => { createReelContent(reel); positions[index] = 0; intervals[index] = setInterval(() => { positions[index] = (positions[index] + 1) % 30; reel.style.transform = `translateY(${-positions[index] * 100}px)`; }, 60); }); } function stopReel(index) { if (!spinning || stopped[index]) return; clearInterval(intervals[index]); stopped[index] = true; const pos = positions[index]; reels[index].style.transform = `translateY(${-pos * 100}px)`; if (stopped.every(v => v)) { spinning = false; checkResult(); } } function checkResult() { const results = positions.map(pos => symbols[pos % symbols.length]); const resultText = document.getElementById("result"); if (results[0] === results[1] && results[1] === results[2]) { if (results[0] === "seven.jpg") { coins += 50; resultText.textContent = "🎉 大当たり！+50コイン 🎉"; } else { coins += 15; resultText.textContent = "✨ 当たり！+15コイン ✨"; } } else { resultText.textContent = "😢 ハズレ… -3コイン"; } updateCoinDisplay(); if (coins < 3) { resultText.textContent += "\nコインが足りません。リロードで再スタート。"; document.getElementById("spinBtn").disabled = true; } } function updateCoinDisplay() { document.getElementById("coinDisplay").textContent = coins; document.getElementById("spinBtn").disabled = coins < 3 ? true : false; }
